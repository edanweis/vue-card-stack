      # Basic

      import Vue from 'vue'
import VueCardStack from 'vue-card-stack'

export default {
  components: {
    VueCardStack
  },
  data() {
    return {
      cards: [
        { background: "#00659d" },
        { background: "#00abbc" },
        { background: "#e2c58a" },
        { background: "#fc8890" },
        { background: "#b35d7f" }
      ]
    }
  }
}

      <vue-card-stack :cards="cards" :stack-width="360" :card-width="280">
  <template v-slot:card="{ card }">
    <div 
      style="width: 100%; height: 100%;"
      :style="{ background: card.background }"
    >
    </div>
  </template>
</vue-card-stack>
